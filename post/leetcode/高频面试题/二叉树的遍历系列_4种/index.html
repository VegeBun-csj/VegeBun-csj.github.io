题目描述

对二叉树进行前中后层序遍历，分别递归与非递归遍历

思路

  • 递归遍历

    三种遍历的唯一区别就是对根节点的操作和递归操作左右子树的顺序不一样

  • 迭代遍历（重要）

    迭代遍历写的比较少，需要重点关注，在面试题中经常出现。对于前中后序遍历，通常采用栈；对于层次遍历采用队列

class Solution {
public:

    void pre(TreeNode* root,vector<int>& res){
        if(root == nullptr) return;
        res.push_back(root -> val);	//根
        pre(root -> left,res);		//左
        pre(root -> right,res);		//右
    }

    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        pre(root,res);
        return res;
    }
};

class Solution {
public:

    vector<vector<int>> res;
    void DFS(TreeNode* root,int dep){
        if(! root) return;
        if(dep >= res.size()) {
            res.push_back(vector<int>());
        }
        res[dep].push_back(root -> val);

        DFS(root -> left,dep + 1);
        DFS(root -> right,dep + 1);
    }

    vector<vector<int>> levelOrder(TreeNode* root) {
        DFS(root,0);
        return res;
    }
};

复杂度分析

因为是递归的遍历，所以时间复杂度为 O(n)，空间复杂度为O(log N)，即树的高度

class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> res;
        if(! root) return res;
        stack<TreeNode*> stk;
        TreeNode* temp = root;
        stk.push(root);	//根入栈
        while(! stk.empty()){
            temp = stk.top();	//取出根
            stk.pop();
            res.push_back(temp -> val);
            if(temp -> right) stk.push(temp -> right);	//右
            if(temp -> left) stk.push(temp -> left);
        }
        return res;
    }
};

中序遍历

使用辅助栈进行存储，中序遍历是左根右，所以遍历的时候，先将根节点入栈，将根节点的左节点入栈，然后将左节点的左节点入栈………最终将所有的最左边的节点都入栈。然后依次弹出，先弹出的是最左边的节点,然后判断该节点是否有右节点，如果有那么就遍历这个节点的右子树，再进入while循环

class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> res;
        if(! root) return res;
        stack<TreeNode*> stk;
        TreeNode* temp = root;
        
        //注意这里必须要加temp的判断条件，因为遍历完左子树(包括根节点)之后，此时栈为空，而temp此时可能是指向了根节点的右孩子，此时还需要进入寻循环的
        while(! stk.empty() || temp){
            while(temp){
                stk.push(temp);
                temp = temp -> left;
            }
            TreeNode* node = stk.top();
            stk.pop();
            res.push_back(node -> val);

            if(node -> right){
                temp = node -> right;
            }
        }
        return res;
    }
};

后序遍历

采用辅助栈进行存储，后序遍历的顺序为左右根，所以对于栈来讲，我们肯定要得到一个根右左的顺序，再输出才能满足，所以这里采用两个栈

第一个栈stk1的入栈顺序为根左右，也就是压栈的时候先压入左子树，再压入右子树；然后以根右左的顺序输入到第二个栈stk2，再利用栈的先进后出的特点，输出栈stk2中元素，最终得到的顺序就是左右根

class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> res;
        stack<TreeNode*> stk1;
        stack<TreeNode*> stk2;
        if(! root) return res;

        TreeNode* temp = root;
        stk1.push(root);
        while(! stk1.empty()){
            temp = stk1.top();
            stk1.pop();
            //以根右左的顺序输入到stk2(即以根左右的顺序入栈stk1)
            stk2.push(temp);
            if(temp -> left) stk1.push(temp -> left);
            if(temp -> right) stk1.push(temp -> right);
        }

        //以左右根的顺序输出
        while(! stk2.empty()) {
            res.push_back(stk2.top() -> val);
            stk2.pop();
        }
        
        return res;
    }
};

class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        queue<TreeNode*> q;
        vector<vector<int>> res;
        if(! root) return res;
        TreeNode* temp = root;
        q.push(root);
        while(! q.empty()){
            int levelSize = q.size();	//队列中节点数即位每一层的节点数目
            res.push_back(vector<int> ());
            
            for(int i = 0;i < levelSize;i ++){	//将当前层的节点全部出队
                temp = q.front();       //记录队首节点
                q.pop();
                res.back().push_back(temp -> val);	//用vector记录每层的节点
                if(temp -> left) q.push(temp -> left);	//每出一个当前层的节点，就将其左右孩子节点入队
                if(temp -> right) q.push(temp -> right);
            }
        }
        return res;
    }
};

复杂度分析

前三种遍历：时间复杂度为 O(n)，空间复杂度为O(h)

层次遍历：时间复杂度为O(n)，空间复杂度为O(N)，因为基本上每个节点都遍历了，即每一层的节点
